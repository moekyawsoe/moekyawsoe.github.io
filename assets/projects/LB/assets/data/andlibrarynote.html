MainActivity.java

package mks.dev.librarytest.ui;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.Toast;

import mks.dev.andlibrarynote.MksToast;
import mks.dev.librarytest.R;

public class AndLibraryNote extends AppCompatActivity implements View.OnClickListener {

    Button btn_success,btn_info,btn_error,btn_warning,btn_csone,btn_cstwo,btn_csthree,btn_default;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_and_library_note);
        btn_success = findViewById(R.id.btn_success);
        btn_info = findViewById(R.id.btn_info);
        btn_error = findViewById(R.id.btn_error);
        btn_warning = findViewById(R.id.btn_warning);
        btn_csone = findViewById(R.id.btn_csone);
        btn_cstwo = findViewById(R.id.btn_cstwo);
        btn_csthree = findViewById(R.id.btn_csthree);
        btn_default = findViewById(R.id.btn_default);

        btn_success.setOnClickListener(this);
        btn_info.setOnClickListener(this);
        btn_error.setOnClickListener(this);
        btn_warning.setOnClickListener(this);
        btn_csone.setOnClickListener(this);
        btn_cstwo.setOnClickListener(this);
        btn_csthree.setOnClickListener(this);
        btn_default.setOnClickListener(this);

    }

    @Override
    public void onClick(View v) {
        switch (v.getId()){
            case R.id.btn_csone:
                MksToast.makeText(getApplicationContext(),"This is Toast Without Badge", Toast.LENGTH_LONG,MksToast.Default,false).show();
                break;
            case R.id.btn_csthree:
                MksToast.makeText(getApplicationContext(),"This is Custom Badge & Corner icon Toast", Toast.LENGTH_LONG, MksToast.INFO,R.drawable.logopng,R.drawable.logopng).show();
                break;
            case R.id.btn_cstwo:
                MksToast.makeText(getApplicationContext(),"This is Custom Badge Toast", Toast.LENGTH_LONG, MksToast.INFO,R.drawable.logopng,true).show();
                break;
            case R.id.btn_default:
                MksToast.makeText(getApplicationContext(),"This is Default Toast", Toast.LENGTH_LONG,MksToast.Default,true).show();
                break;
            case R.id.btn_error:
                MksToast.makeText(getApplicationContext(),"This is Error Toast", Toast.LENGTH_LONG,MksToast.ERROR,true).show();
                break;
            case R.id.btn_info:
                MksToast.makeText(getApplicationContext(),"This is Info Toast", Toast.LENGTH_LONG,MksToast.INFO,true).show();
                break;
            case R.id.btn_success:
                MksToast.makeText(getApplicationContext(),"This is Success Toast", Toast.LENGTH_LONG,MksToast.SUCCESS,true).show();
                break;
            case R.id.btn_warning:
                MksToast.makeText(getApplicationContext(),"This is Warning Toast", Toast.LENGTH_LONG,MksToast.WARNING,true).show();
                break;
        }
    }
}
